window.initStoreAddressAutocomplete = function () { var addressInput = document.getElementById('mainForm:storeAddress'); var pinCodeInput = document.getElementById('mainForm:storePinCode_input') || document.getElementById('mainForm:storePinCode'); if (!addressInput || !window.google || !google.maps || !google.maps.places) { return; } var autocomplete = new google.maps.places.Autocomplete(addressInput, {types: ['address']}); autocomplete.addListener('place_changed', function () { var place = autocomplete.getPlace(); if (!place) { return; } if (place.formatted_address) { addressInput.value = place.formatted_address; addressInput.dispatchEvent(new Event('input', {bubbles: true})); addressInput.dispatchEvent(new Event('change', {bubbles: true})); } if (!pinCodeInput || !place.address_components) { return; } var postal = place.address_components.find(function (component) { return component.types && component.types.indexOf('postal_code') !== -1; }); if (postal) { pinCodeInput.value = postal.long_name; pinCodeInput.dispatchEvent(new Event('input', {bubbles: true})); pinCodeInput.dispatchEvent(new Event('change', {bubbles: true})); } }); };
